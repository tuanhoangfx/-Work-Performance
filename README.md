# Infi Project - Task Manager

## üìñ Project Overview

Infi Project Task Manager is a modern, elegant internal tool designed for managing team member tasks through an intuitive, Kanban-style interface. It focuses on a smooth user experience, clear data presentation, and a dedicated dashboard for managers.

## ‚ú® Key Features

*   **Kanban Board**: An interactive board for team members to view and manage their tasks, organized by status ('To Do', 'In Progress', 'Done').
*   **Task Management**: Users can create, update, and delete their own tasks. Each task can have a title, description, status, priority, and due date.
*   **Task Comments**: A full commenting system on each task to facilitate team communication.
*   **File Attachments**: Attach files (images, documents) to tasks for better context and information sharing, with support for direct pasting from the clipboard.
*   **Real-time Notifications**: Get notified instantly when you're assigned a task or when someone comments on your tasks.
*   **Priority and Status**: Clearly defined priorities ('Low', 'Medium', 'High') and statuses help in organizing and tracking work efficiently.
*   **Comprehensive Manager Dashboard**:
    *   A separate view for users with an 'admin' role.
    *   View a list of all team members and select any team member to view their task board.
    *   Full CRUD (Create, Read, Update, Delete) functionality for tasks for any team member.
    *   Includes an 'Overall Summary' view showing aggregated statistics for all tasks in the system.
*   **Secure Authentication & Authorization**: Powered by Supabase Auth with Row Level Security (RLS) to ensure users can only access their own data, while managers have full access.
*   **Customizable UI**: Switch between Light/Dark modes with beautiful gradient color themes.
*   **Multi-language Support**: English, Vietnamese (Ti·∫øng Vi·ªát), and Thai (‡πÑ‡∏ó‡∏¢).
*   **Responsive Design**: A fully responsive layout for all screen sizes.
*   **In-App User Guide**: A comprehensive guide available directly within the application.

## üíª Tech Stack

-   **Frontend**: React, TypeScript, Tailwind CSS
-   **Backend**: Supabase (Database, Auth, Storage, RLS, Realtime)
-   **State Management**: React Context API & Component State

```sql
-- moddatetime
CREATE OR REPLACE FUNCTION public.moddatetime()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = now();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'handle_updated_at') THEN
    CREATE TRIGGER handle_updated_at
      BEFORE UPDATE ON public.tasks
      FOR EACH ROW EXECUTE FUNCTION public.moddatetime();
  END IF;
END $$;

-- handle_new_user
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER SET search_path = public
AS $$
BEGIN
  INSERT INTO public.profiles (id, full_name, avatar_url, role, created_at, last_sign_in_at)
  VALUES (NEW.id, NEW.raw_user_meta_data->>'full_name', NEW.raw_user_meta_data->>'avatar_url', 'employee', NEW.created_at, NEW.last_sign_in_at)
  ON CONFLICT (id) DO NOTHING;
  RETURN NEW;
END;
$$;

DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'on_auth_user_created') THEN
    CREATE TRIGGER on_auth_user_created
      AFTER INSERT ON auth.users
      FOR EACH ROW EXECUTE FUNCTION public.handle_new_user();
  END IF;
END $$;

-- handle_user_update for last_sign_in_at
CREATE OR REPLACE FUNCTION public.handle_user_update()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER SET search_path = public
AS $$
BEGIN
  UPDATE public.profiles
  SET last_sign_in_at = NEW.last_sign_in_at
  WHERE id = NEW.id;
  RETURN NEW;
END;
$$;

DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'on_auth_user_updated') THEN
    CREATE TRIGGER on_auth_user_updated
      AFTER UPDATE ON auth.users
      FOR EACH ROW
      WHEN (OLD.last_sign_in_at IS DISTINCT FROM NEW.last_sign_in_at)
      EXECUTE FUNCTION public.handle_user_update();
  END IF;
END $$;

-- can_access_task_from_attachment_path
CREATE OR REPLACE FUNCTION public.can_access_task_from_attachment_path(p_file_path text)
RETURNS boolean
LANGUAGE plpgsql
SECURITY DEFINER SET search_path = public
AS $$
DECLARE
  v_task_id bigint;
  v_has_access boolean := false;
BEGIN
  BEGIN
    v_task_id := (string_to_array(p_file_path, '/'))[2]::bigint;
  EXCEPTION WHEN OTHERS THEN
    RETURN FALSE;
  END;
  IF v_task_id IS NULL THEN RETURN FALSE; END IF;
  SELECT EXISTS (
    SELECT 1 FROM public.tasks
    WHERE id = v_task_id AND (user_id = auth.uid() OR created_by = auth.uid())
  ) INTO v_has_access;
  RETURN COALESCE(v_has_access, FALSE);
END;
$$;

-- is_project_member
CREATE OR REPLACE FUNCTION public.is_project_member(p_project_id bigint, p_user_id uuid)
RETURNS boolean
LANGUAGE plpgsql
SECURITY DEFINER SET search_path = public
AS $$
BEGIN
  RETURN EXISTS (
    SELECT 1 FROM public.project_members
    WHERE project_id = p_project_id AND user_id = p_user_id
  );
END;
$$;

-- create_new_task_assignment_notification
CREATE OR REPLACE FUNCTION public.create_new_task_assignment_notification()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER SET search_path = public
AS $$
DECLARE
  assigner_name text;
BEGIN
  SELECT full_name INTO assigner_name FROM public.profiles WHERE id = auth.uid();
  IF NEW.user_id IS NOT NULL AND NEW.user_id <> auth.uid() THEN
    INSERT INTO public.notifications (user_id, actor_id, type, data)
    VALUES (
      NEW.user_id, auth.uid(), 'new_task_assigned',
      jsonb_build_object('task_id', NEW.id, 'task_title', NEW.title, 'assigner_name', assigner_name)
    );
  END IF;
  RETURN NEW;
END;
$$;

DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'on_new_task_created_for_other') THEN
    CREATE TRIGGER on_new_task_created_for_other
      AFTER INSERT ON public.tasks
      FOR EACH ROW EXECUTE FUNCTION public.create_new_task_assignment_notification();
  END IF;
  IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'on_task_reassigned') THEN
    CREATE TRIGGER on_task_reassigned
      AFTER UPDATE OF user_id ON public.tasks
      FOR EACH ROW
      WHEN (OLD.user_id IS DISTINCT FROM NEW.user_id)
      EXECUTE FUNCTION public.create_new_task_assignment_notification();
  END IF;
END $$;

-- create_new_comment_notification
CREATE OR REPLACE FUNCTION public.create_new_comment_notification()
RETURNS TRIGGER
LANGUAGE plpgsql
SECURITY DEFINER SET search_path = public
AS $$
DECLARE
  task_info RECORD;
  commenter_name text;
BEGIN
  SELECT user_id, created_by, title INTO task_info FROM public.tasks WHERE id = NEW.task_id;
  SELECT full_name INTO commenter_name FROM public.profiles WHERE id = NEW.user_id;

  IF task_info.user_id IS NOT NULL AND task_info.user_id <> NEW.user_id THEN
    INSERT INTO public.notifications (user_id, actor_id, type, data)
    VALUES (task_info.user_id, NEW.user_id, 'new_comment',
            jsonb_build_object('task_id', NEW.task_id, 'task_title', task_info.title, 'commenter_name', commenter_name));
  END IF;

  IF task_info.created_by IS NOT NULL
     AND task_info.created_by <> NEW.user_id
     AND task_info.created_by <> task_info.user_id THEN
    INSERT INTO public.notifications (user_id, actor_id, type, data)
    VALUES (task_info.created_by, NEW.user_id, 'new_comment',
            jsonb_build_object('task_id', NEW.task_id, 'task_title', task_info.title, 'commenter_name', commenter_name));
  END IF;

  RETURN NEW;
END;
$$;

DO $$
BEGIN
  IF NOT EXISTS (SELECT 1 FROM pg_trigger WHERE tgname = 'on_new_comment') THEN
    CREATE TRIGGER on_new_comment
      AFTER INSERT ON public.task_comments
      FOR EACH ROW EXECUTE FUNCTION public.create_new_comment_notification();
  END IF;
END $$;

CREATE POLICY "Users can upload task attachments." ON storage.objects
  FOR INSERT WITH CHECK (
    bucket_id = 'task-attachments'
    AND auth.uid()::text = (storage.foldername(name))[1]
    AND public.can_access_task_from_attachment_path(name)
  );

DROP POLICY IF EXISTS "Users can delete their own task attachments." ON storage.objects;
CREATE POLICY "Users can delete their own task attachments." ON storage.objects
  FOR DELETE USING (
    bucket_id = 'task-attachments'
    AND auth.uid()::text = (storage.foldername(name))[1]
    AND public.can_access_task_from_attachment_path(name)
  );

DROP POLICY IF EXISTS "Admins can manage all attachments." ON storage.objects;
CREATE POLICY "Admins can manage all attachments." ON storage.objects
  FOR ALL USING (bucket_id = 'task-attachments' AND public.is_admin())
  WITH CHECK (bucket_id = 'task-attachments' AND public.is_admin());
```
